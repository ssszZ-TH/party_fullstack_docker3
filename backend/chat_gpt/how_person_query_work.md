```sql
-- คำสั่ง SQL นี้ใช้ดึงข้อมูลเกี่ยวกับบุคคลจากหลายตารางที่เกี่ยวข้อง
-- รวมข้อมูลจากตาราง person, gender_type, personname, maritalstatus, physicalcharacteristic, citizenship, และ country
-- ใช้ LEFT JOIN เพื่อให้ได้ข้อมูลบุคคลทุกคน แม้ว่าจะไม่มีข้อมูลในตารางที่เกี่ยวข้อง
-- มีการใช้ subquery เพื่อกรองข้อมูลเฉพาะ เช่น ชื่อตัว, ความสูง, และสัญชาติล่าสุด
SELECT 
    -- เลือกคอลัมน์จากตาราง person
    p.id,                                    -- รหัสประจำตัวของบุคคล (คีย์หลัก)
    p.personal_id_number,                    -- เลขประจำตัวประชาชน (เช่น เลขบัตร 13 หลักของไทย)
    p.birthdate,                             -- วันเกิดของบุคคล
    p.mothermaidenname,                      -- นามสกุลเดิมของมารดา (ใช้ระบุตัวตนในบางบริบท)
    p.totalyearworkexperience,               -- จำนวนปีประสบการณ์ทำงานทั้งหมด
    p.comment,                               -- ความคิดเห็นเพิ่มเติมเกี่ยวกับบุคคล (เช่น อาชีพ, สัญชาติ)
    p.gender_type_id,                        -- รหัสประเภทเพศ (คีย์ต่างประเทศอ้างถึงตาราง gender_type เช่น 1=ชาย, 2=หญิง, 3=ไม่ระบุ)
    
    -- เลือกคำอธิบายเพศจากตาราง gender_type
    gt.description AS gender_description,    -- คำอธิบายประเภทเพศ (เช่น "ชาย", "หญิง", "ไม่ระบุเพศ")

    -- เลือกรายละเอียดชื่อตัวจากตาราง personname
    pn1.id AS fname_id,                      -- รหัสประจำตัวของชื่อตัว
    pn1.name AS fname,                       -- ชื่อตัวของบุคคล
    pn1.fromdate AS fname_fromdate,          -- วันที่เริ่มใช้ชื่อตัว
    pn1.thrudate AS fname_thrudate,          -- วันที่เลิกใช้ชื่อตัว (NULL ถ้ายังใช้อยู่)
    pn1.personnametype_id AS fname_personnametype_id, -- รหัสประเภทชื่อสำหรับชื่อตัว (อ้างถึงตาราง personnametype)
    pnt1.description AS fname_personnametype_description, -- คำอธิบายประเภทชื่อ (เช่น "FirstName")

    -- เลือกรายละเอียดชื่อกลางจากตาราง personname
    pn2.id AS mname_id,                      -- รหัสประจำตัวของชื่อกลาง
    pn2.name AS mname,                       -- ชื่อกลางของบุคคล
    pn2.fromdate AS mname_fromdate,          -- วันที่เริ่มใช้ชื่อกลาง
    pn2.thrudate AS mname_thrudate,          -- วันที่เลิกใช้ชื่อกลาง (NULL ถ้ายังใช้อยู่)
    pn2.personnametype_id AS mname_personnametype_id, -- รหัสประเภทชื่อสำหรับชื่อกลาง (อ้างถึงตาราง personnametype)
    pnt2.description AS mname_personnametype_description, -- คำอธิบายประเภทชื่อ (เช่น "MiddleName")

    -- เลือกรายละเอียดนามสกุลจากตาราง personname
    pn3.id AS lname_id,                      -- รหัสประจำตัวของนามสกุล
    pn3.name AS lname,                       -- นามสกุลของบุคคล
    pn3.fromdate AS lname_fromdate,          -- วันที่เริ่มใช้นามสกุล
    pn3.thrudate AS lname_thrudate,          -- วันที่เลิกใช้นามสกุล (NULL ถ้ายังใช้อยู่)
    pn3.personnametype_id AS lname_personnametype_id, -- รหัสประเภทชื่อสำหรับนามสกุล (อ้างถึงตาราง personnametype)
    pnt3.description AS lname_personnametype_description, -- คำอธิบายประเภทชื่อ (เช่น "LastName")

    -- เลือกรายละเอียดสถานภาพสมรสจากตาราง maritalstatus
    ms.id AS marital_status_id,              -- รหัสประจำตัวของสถานภาพสมรส
    ms.fromdate AS marital_status_fromdate,  -- วันที่เริ่มสถานภาพสมรส
    ms.thrudate AS marital_status_thrudate,  -- วันที่สิ้นสุดสถานภาพสมรส (NULL ถ้ายังมีผล)
    ms.maritalstatustype_id AS marital_status_type_id, -- รหัสประเภทสถานภาพสมรส (อ้างถึงตาราง maritalstatustype)
    mst.description AS marital_status_type_description, -- คำอธิบายสถานภาพสมรส (เช่น "โสด", "สมรส")

    -- เลือกรายละเอียดความสูงจากตาราง physicalcharacteristic
    pc1.id AS height_id,                     -- รหัสประจำตัวของข้อมูลความสูง
    pc1.val AS height_val,                   -- ค่าความสูง (เช่น เป็นเซนติเมตร)
    pc1.fromdate AS height_fromdate,         -- วันที่บันทึกความสูง
    pc1.thrudate AS height_thrudate,         -- วันที่ข้อมูลความสูงเลิกใช้ (NULL ถ้ายังมีผล)
    pc1.physicalcharacteristictype_id AS height_type_id, -- รหัสประเภทลักษณะกายภาพสำหรับความสูง
    pct1.description AS height_type_description, -- คำอธิบายประเภทลักษณะกายภาพ (เช่น "Height")

    -- เลือกรายละเอียดน้ำหนักจากตาราง physicalcharacteristic
    pc2.id AS weight_id,                     -- รหัสประจำตัวของข้อมูลน้ำหนัก
    pc2.val AS weight_val,                   -- ค่าน้ำหนัก (เช่น เป็นกิโลกรัม)
    pc2.fromdate AS weight_fromdate,         -- วันที่บันทึกน้ำหนัก
    pc2.thrudate AS weight_thrudate,         -- วันที่ข้อมูลน้ำหนักเลิกใช้ (NULL ถ้ายังมีผล)
    pc2.physicalcharacteristictype_id AS weight_type_id, -- รหัสประเภทลักษณะกายภาพสำหรับน้ำหนัก
    pct2.description AS weight_type_description, -- คำอธิบายประเภทลักษณะกายภาพ (เช่น "Weight")

    -- เลือกรายละเอียดสัญชาติจากตาราง citizenship
    c.id AS citizenship_id,                  -- รหัสประจำตัวของข้อมูลสัญชาติ
    c.fromdate AS citizenship_fromdate,      -- วันที่เริ่มมีสัญชาติ
    c.thrudate AS citizenship_thrudate,      -- วันที่สิ้นสุดสัญชาติ (NULL ถ้ายังมีผล)
    c.country_id AS country_id,              -- รหัสประเทศ (คีย์ต่างประเทศอ้างถึงตาราง country)
    co.isocode AS country_isocode,           -- รหัส ISO 3166-1 alpha-2 ของประเทศ (เช่น "TH", "US")
    co.name_en AS country_name_en,           -- ชื่อประเทศเป็นภาษาอังกฤษ
    co.name_th AS country_name_th            -- ชื่อประเทศเป็นภาษาไทย

FROM person p
    -- เชื่อมตาราง gender_type เพื่อดึงคำอธิบายเพศ
    LEFT JOIN gender_type gt ON p.gender_type_id = gt.id
        -- LEFT JOIN ทำให้ได้ข้อมูลบุคคลทุกคน แม้ว่าจะไม่มี gender_type_id
        -- จับคู่ p.gender_type_id กับ gt.id

    -- เชื่อมตาราง personname เพื่อดึงชื่อตัว
    LEFT JOIN personname pn1 ON pn1.person_id = p.id 
        AND pn1.personnametype_id = (SELECT id FROM personnametype WHERE description = 'FirstName')
        -- กรองเฉพาะชื่อที่มี personnametype เป็น "FirstName"
        -- Subquery ดึง id ของ personnametype ที่ description เป็น "FirstName"
        -- LEFT JOIN ทำให้ได้บุคคลที่ไม่มีชื่อตัวด้วย

    -- เชื่อมตาราง personnametype เพื่อดึงคำอธิบายประเภทชื่อตัว
    LEFT JOIN personnametype pnt1 ON pn1.personnametype_id = pnt1.id
        -- จับคู่ personname.personnametype_id กับ personnametype.id
        -- ได้คำอธิบายประเภทชื่อ (เช่น "FirstName")

    -- เชื่อมตาราง personname เพื่อดึงชื่อกลาง
    LEFT JOIN personname pn2 ON pn2.person_id = p.id 
        AND pn2.personnametype_id = (SELECT id FROM personnametype WHERE description = 'MiddleName')
        -- กรองเฉพาะชื่อที่มี personnametype เป็น "MiddleName"
        -- Subquery ดึง id ของ personnametype ที่ description เป็น "MiddleName"
        -- LEFT JOIN ทำให้ได้บุคคลที่ไม่มีชื่อกลางด้วย

    -- เชื่อมตาราง personnametype เพื่อดึงคำอธิบายประเภทชื่อกลาง
    LEFT JOIN personnametype pnt2 ON pn2.personnametype_id = pnt2.id
        -- จับคู่ personname.personnametype_id กับ personnametype.id
        -- ได้คำอธิบายประเภทชื่อ (เช่น "MiddleName")

    -- เชื่อมตาราง personname เพื่อดึงนามสกุล
    LEFT JOIN personname pn3 ON pn3.person_id = p.id 
        AND pn3.personnametype_id = (SELECT id FROM personnametype WHERE description = 'LastName')
        -- กรองเฉพาะชื่อที่มี personnametype เป็น "LastName"
        -- Subquery ดึง id ของ personnametype ที่ description เป็น "LastName"
        -- LEFT JOIN ทำให้ได้บุคคลที่ไม่มีนามสกุลด้วย

    -- เชื่อมตาราง personnametype เพื่อดึงคำอธิบายประเภทนามสกุล
    LEFT JOIN personnametype pnt3 ON pn3.personnametype_id = pnt3.id
        -- จับคู่ personname.personnametype_id กับ personnametype.id
        -- ได้คำอธิบายประเภทชื่อ (เช่น "LastName")

    -- เชื่อมตาราง maritalstatus เพื่อดึงสถานภาพสมรส
    LEFT JOIN maritalstatus ms ON ms.person_id = p.id
        -- จับคู่ maritalstatus.person_id กับ person.id
        -- LEFT JOIN ทำให้ได้บุคคลที่ไม่มีข้อมูลสถานภาพสมรสด้วย

    -- เชื่อมตาราง maritalstatustype เพื่อดึงคำอธิบายสถานภาพสมรส
    LEFT JOIN maritalstatustype mst ON ms.maritalstatustype_id = mst.id
        -- จับคู่ maritalstatus.maritalstatustype_id กับ maritalstatustype.id
        -- ได้คำอธิบายสถานภาพสมรส (เช่น "โสด", "สมรส")

    -- เชื่อมตาราง physicalcharacteristic เพื่อดึงความสูง
    LEFT JOIN physicalcharacteristic pc1 ON pc1.person_id = p.id 
        AND pc1.physicalcharacteristictype_id = (SELECT id FROM physicalcharacteristictype WHERE description = 'Height')
        -- กรองเฉพาะข้อมูลที่มี physicalcharacteristictype เป็น "Height"
        -- Subquery ดึง id ของ physicalcharacteristictype ที่ description เป็น "Height"
        -- LEFT JOIN ทำให้ได้บุคคลที่ไม่มีข้อมูลความสูงด้วย

    -- เชื่อมตาราง physicalcharacteristictype เพื่อดึงคำอธิบายประเภทความสูง
    LEFT JOIN physicalcharacteristictype pct1 ON pc1.physicalcharacteristictype_id = pct1.id
        -- จับคู่ physicalcharacteristic.physicalcharacteristictype_id กับ physicalcharacteristictype.id
        -- ได้คำอธิบายประเภทลักษณะกายภาพ (เช่น "Height")

    -- เชื่อมตาราง physicalcharacteristic เพื่อดึงน้ำหนัก
    LEFT JOIN physicalcharacteristic pc2 ON pc2.person_id = p.id 
        AND pc2.physicalcharacteristictype_id = (SELECT id FROM physicalcharacteristictype WHERE description = 'Weight')
        -- กรองเฉพาะข้อมูลที่มี physicalcharacteristictype เป็น "Weight"
        -- Subquery ดึง id ของ physicalcharacteristictype ที่ description เป็น "Weight"
        -- LEFT JOIN ทำให้ได้บุคคลที่ไม่มีข้อมูลน้ำหนักด้วย

    -- เชื่อมตาราง physicalcharacteristictype เพื่อดึงคำอธิบายประเภทน้ำหนัก
    LEFT JOIN physicalcharacteristictype pct2 ON pc2.physicalcharacteristictype_id = pct2.id
        -- จับคู่ physicalcharacteristic.physicalcharacteristictype_id กับ physicalcharacteristictype.id
        -- ได้คำอธิบายประเภทความยาว (เช่น "Weight")

    -- เชื่อมตาราง citizenship เพื่อดึงข้อมูลสัญชาติล่าสุด
    LEFT JOIN (
        SELECT c.* 
        FROM citizenship c
        ORDER BY c.fromdate DESC LIMIT 1
        -- Subquery เลือกข้อมูลทั้งหมดจากตาราง citizenship
        -- ORDER BY c.fromdate DESC จัดเรียงตามวันที่เริ่มจากล่าสุด
        -- LIMIT 1 เลือกเฉพาะสัญชาติล่าสุดของบุคคล
    ) c ON c.person_id = p.id
        -- จับคู่ citizenship.person_id กับ person.id
        -- LEFT JOIN ทำให้ได้บุคคลที่ไม่มีข้อมูลสัญชาติด้วย

    -- เชื่อมตาราง country เพื่อดึงข้อมูลชื่อประเทศ
    LEFT JOIN country co ON c.country_id = co.id
        -- จับคู่ citizenship.country_id กับ country.id
        -- ได้ข้อมูลเช่น รหัส ISO และชื่อ
        -- LEFT JOIN ทำให้ได้บุคคลที่ไม่มีข้อมูลสัญชาติหรือประเทศด้วย

-- จัดเรียงผลลัพธ์ตามรหัส id ของบุคคลจากน้อยไปมาก
ORDER BY p.id ASC
    -- ทำให้ผลลัพท์เรียงตาม id เพื่อให้ดูง่ายและเป็นระเบียบ
```
